```c
// 可以利用地址“隐含”地返回值 &min返回最小值
int DeleteMin(List &L, int &min){
	if(L.length == 0)
        return -1;
    int last = L.data[L.length-1];  // 存储最后一个元素
    min = L.data[0];
    int pos;  // 最小元素的位置
    for(int i = 0; i < L.length; i ++){
        if(L.data[i] < min){
            min = L.data[i];
            pos = i;   // i会一直增加
        }
    }
    L.data[i] = last;
    return pos;  // 返回最小值位置
}
```

```c
int UpSet(List &L){
	if(L.length == 0)
        return -1;
    int temp;  // 中间值
    for(int i=0, j=L.length-1; i<=j;i++, j--){
        temp = L.data[i];
        L.data[i] = L.data[j];
        L.data[j] = temp;
    }
    return 1;
}
```

```c
int eleX(List &L, int X){
	if(L.length == 0)
        return -2;
    int len = L.length;
    for(int i = 0; i < len; i ++){
        if(L.data[i] == X){
            L.data[i] = L.data[L.length-1];  // 用最后一个元素来代替
            len --;  // 长度减一
        }
    }
    L.length = len;
    return 1;
}
```

```c
int Deletest(List &L, int s, int t){
	if(L.length == 0)
        return -2;
    if(s > t)
        return -1;
    int i; // 第一个大于s的元素位置
    int j; // 第一个大于等于t的元素位置
    
    int len = L.length;
    for(i=0; i < len; i ++)   // 找到i
        if(L.data[i] > s)
            break;
    for(j=i;j < len; j ++)  // 找到j
        if(L.data[j] >= t)
            break;
    for(; i < len && j < len; i ++, j ++){
        L.data[i] = L.data[j];   // 将后面的元素代替在st范围内的元素
    }
    L.length = i;  // z
    return 1;
}
```
